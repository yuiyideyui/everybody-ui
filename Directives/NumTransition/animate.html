指令参数 ​

ts
interface TransitionOptions {
  value: number; //值
  duration?: number; //动画时长 默认1000ms
  transition?: TransitionCurve; //动画曲线 默认easeInOutCubic - >https://easings.net/ 
  format?: boolean; // 是否千分位格式化 默认false
}
